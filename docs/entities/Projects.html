import { Column, Entity } from "typeorm";

@Entity("projects", { schema: "public" })
export class Projects {
  @Column("uuid", {
    primary: true,
    name: "uuid",
    default: () => "uuid_generate_v4()",
  })
  uuid!: string;

  @Column("character varying", { name: "name", length: 150 })
  name!: string;

  @Column("text", { name: "description" })
  description!: string;

  @Column("character varying", {
    name: "engine_file",
    nullable: true,
    length: 250,
  })
  engine_file!: string | null;

  @Column("character varying", {
    name: "context_file",
    nullable: true,
    length: 250,
  })
  context_file!: string | null;

  @Column("boolean", { name: "isComplete", default: () => "true" })
  isComplete!: boolean;

  @Column("timestamp without time zone", {
    name: "created_at",
    default: () => "'2023-05-29 19:08:01.935'",
  })
  created_at!: Date;

  @Column("timestamp without time zone", {
    name: "updated_at",
    default: () => "'2023-05-29 19:08:01.935'",
  })
  updated_at!: Date;
}
